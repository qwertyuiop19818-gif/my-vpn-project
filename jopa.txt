-- Функция переключения видимости панели
local function toggleMenu()
	MainFrame.Visible = not MainFrame.Visible
end

-- 1. Открытие по кнопке на клавиатуре (для ПК)
UserInputService.InputBegan:Connect(function(input, gameProcessed)
	if gameProcessed then return end
	if input.KeyCode == toggleKey then
		toggleMenu()
	end
end)

-- 2. Открытие по тапу на круглую кнопку (для мобилок и ПК)
MobileToggleButton.MouseButton1Click:Connect(function()
	toggleMenu()
end)

-- Проверка устройства: если это ПК, то круглую кнопку можно оставить или скрыть. 
-- Оставим её активной для всех, так как это удобная альтернатива кликанью клавиатуры.

-- Функция обновления хитбоксов
local function updateHitboxes()
	for _, player in ipairs(Players:GetPlayers()) do
		if player ~= LocalPlayer and player.Character then
			local head = player.Character:FindFirstChild("Head")
			if head and head:IsA("BasePart") then
				head.Size = Vector3.new(hitboxSize, hitboxSize, hitboxSize)
				head.CanCollide = false
				head.Transparency = 0.5
			end
		end
	end
end

-- Цикл для ВХ и Хитбоксов
task.spawn(function()
	while true do
		task.wait(0.5)
		updateHitboxes()
		
		for _, player in ipairs(Players:GetPlayers()) do
			if player ~= LocalPlayer and player.Character then
				local highlight = player.Character:FindFirstChild("ESPHighlight")
				
				if espEnabled then
					if not highlight then
						highlight = Instance.new("Highlight")
						highlight.Name = "ESPHighlight"
						highlight.FillColor = Color3.fromRGB(255, 0, 0)
						highlight.OutlineColor = Color3.fromRGB(255, 255, 255)
						highlight.FillTransparency = 0.5
						highlight.Parent = player.Character
					end
				else
					if highlight then
						highlight:Destroy()
					end
				end
			end
		end
	end
end)

-- Логика кнопок внутри меню
IncreaseHitbox.MouseButton1Click:Connect(function()
	hitboxSize = hitboxSize + 2
	if hitboxSize > 10 then hitboxSize = 10 end
	HitboxLabel.Text = "Размер хитбокса (Head): " .. hitboxSize
	updateHitboxes()
end)

ResetHitbox.MouseButton1Click:Connect(function()
	for _, player in ipairs(Players:GetPlayers()) do
		if player ~= LocalPlayer and player.Character then
			local head = player.Character:FindFirstChild("Head")
			if head then
				head.Size = Vector3.new(2, 1, 1)
				head.Transparency = 0
			end
		end
	end
	hitboxSize = 2
	HitboxLabel.Text = "Размер хитбокса (Head): " .. hitboxSize
end)

EspButton.MouseButton1Click:Connect(function()
	espEnabled = not espEnabled
	if espEnabled then
		EspButton.Text = "ВХ: ВКЛ"
		EspButton.TextColor3 = Color3.fromRGB(50, 255, 50)
	else
		EspButton.Text = "ВХ: ВЫКЛ"
		EspButton.TextColor3 = Color3.fromRGB(255, 50, 50)
		for _, player in ipairs(Players:GetPlayers()) do
			if player.Character and player.Character:FindFirstChild("ESPHighlight") then
				player.Character.ESPHighlight:Destroy()
			end
		end
	end
end)
